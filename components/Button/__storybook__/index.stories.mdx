import { Meta, Story, Canvas } from "@storybook/addon-docs";
import { BUTTON_SIZES, BUTTON_TYPES, BUTTON_VARIANTS } from "../constants";
import Button from "../index";
import { ThemeVariants } from "../../Theme/__storybook__/ThemeVariants.js";

import "./stories.scss";

<Meta
  title="Button"
  component={Button}
  argTypes={{
    size: {
      options: Object.values(BUTTON_SIZES),
      control: { type: "select" },
    },
    type: {
      options: Object.values(BUTTON_TYPES),
      control: { type: "select" },
    },
    variant: Object.values(BUTTON_VARIANTS),
    control: { type: "select" },
  }}
/>

export const Template = ({
  primary,
  type,
  size,
  variant,
  className,
  children = "SOME TEXT",
}) => (
  <Button
    className={className}
    type={type}
    size={size}
    primary={primary}
    variant={variant}
  >
    {children}
  </Button>
);

# Button

This is a building block component for a button.

<Canvas>
  <Story name="default_props">{Template.bind({})}</Story>

## Variants ans sizes

You can choose different variants for the button. Ones should be ["contained", "text", "outlined"]

<Canvas>
  <Story name="variants with sizes">
    <div className="button-variants">
      {Object.entries(BUTTON_VARIANTS).map(([variant_name, variant]) => (
        <div className="button-sizes" key={variant_name}>
          {Object.entries(BUTTON_SIZES).map(([size_name, size]) => (
            <Button key={size_name} size={size} variant={variant}>
              {size_name}
            </Button>
          ))}
        </div>
      ))}
    </div>
  </Story>
</Canvas>
