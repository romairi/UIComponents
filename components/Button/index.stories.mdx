import { Meta, Story, Canvas } from "@storybook/addon-docs";
import Button from "./index";
import { BUTTON_SIZES, BUTTON_TYPES } from "./constants";

import "./stories.scss";

<Meta
  title="Button"
  component={Button}
  argTypes={{
    size: {
      options: Object.values(BUTTON_SIZES),
      control: { type: "select" },
    },
    type: {
      options: Object.values(BUTTON_TYPES),
      control: { type: "select" },
    },
  }}
/>

export const Template = ({
  primary,
  type,
  size,
  className,
  children = "SOME TEXT",
}) => (
  <Button className={className} type={type} size={size} primary={primary}>
    {children}
  </Button>
);

  <Story name="default_props">{Template.bind({})}</Story>

        </Button>

  </Story>
</Canvas>
