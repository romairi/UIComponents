import { Meta, Story, Canvas } from "@storybook/addon-docs";

<Meta

export const Template = ({
  primary,
  type,
  size,
  variant,
  className,
  children = "SOME TEXT",
}) => (
  <Button
    className={className}
    type={type}
    size={size}
    primary={primary}
    variant={variant}

    {children}
  </Button>
);
