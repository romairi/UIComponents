<Meta

  component={TextField}
  argTypes={{

export const Template = ({
  id,
  className,
  theme,
  size,
  text,
  type,
  placeholder,
  variant,
  onChange,
  required,
  autoFocus,
  autoComplete,
  fullWidth,
  error,
}) => (
  <TextField
    id={id}
    className={className}
    type={type}
    size={size}
    text={text}
    placeholder={placeholder}
    variant={variant}
    onChange={onChange}
    required={required}
    autoFocus={autoFocus}
    autoComplete={autoComplete}
    fullWidth={fullWidth}
    error={error}
  />
);

## Variants ans sizes

You can choose different variants for the TextField. Ones should be ["outlined", "filled", "standard"]

<Canvas>
  <Story name="variants with sizes">
    <div className="text-field-variants">
      {Object.entries(TEXT_FIELD_VARIANTS).map(([variant_name, variant]) => (
        <div className="text-field-sizes">
          {Object.entries(TEXT_FIELD_SIZES).map(([size_name, size]) => (
            <TextField
              key={size_name}
              size={size}
              variant={variant}
              placeholder={size_name}
              onChange={onChange}
              required={required}
              autoFocus={autoFocus}
              autoComplete={autoComplete}
              fullWidth={fullWidth}
              error={error}
